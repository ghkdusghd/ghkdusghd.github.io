        for (int i = 2; i * i <= N; i++) {
            if (isPrime[i]) {
                for (int j = i * i; j <= N; j += i) {
                    isPrime[j] = false;
                }

        while (left < primes.size()) {
            if (sum == N) {
                count++;
                sum -= primes.get(left);
                left++;
            } else if (sum < N) {
                if (right == primes.size()) break;
                sum += primes.get(right);
                right++;
            } else if (sum > N) {
                sum -= primes.get(left);
                left++;
            }
        }
        System.out.println(count);
